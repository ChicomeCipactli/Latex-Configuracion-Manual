% --------- Definiciones de teoremas, definiciones, axiomas, problemas...

\definecolor{tscolor}{rgb}{0.6,0,0.3}
\newcommand{\tscolor}{purple}

\newenvironment{proof}[1][\textbf{\textit{Demostración}}]{
     \par\medskip\noindent
     \textcolor{tscolor}{#1.}
     \rmfamily
}{
    $\hfill\textcolor{tscolor}{\blacksquare}$\\
}

\newenvironment{solution}[1][\textbf{\textit{Solución}}]{
     \par\medskip\noindent
     \textcolor{tscolor}{#1.}
     \rmfamily
}{
    $\hfill\textcolor{tscolor}{\square}$\\
}

\newcounter{problem}
\newenvironment{problem}[1][\theproblem]{
    \begin{tcolorbox}[colback=\tscolor!20,colframe=\tscolor!50]
        \refstepcounter{problem}
        \textbf{Problema~#1.}
        \rmfamily
}{
    \end{tcolorbox}
    \medskip
}

\newcounter{lemma}
\newenvironment{lemma}[1][\thelemma]{
    \begin{tcolorbox}[colback=white!20,colframe=\tscolor!50]
        \refstepcounter{lemma}
        \noindent
        \color[rgb]{0.6,0,0.3}
        \textbf{Lema~#1.}
        \rmfamily
}{
    \end{tcolorbox}
    \medskip
}

\newcounter{theorem}
\newenvironment{theorem}[1][\thetheorem]{
    \begin{tcolorbox}[colback=white!20,colframe=\tscolor!50]
        \refstepcounter{theorem}
        \par\medskip
        \noindent
        \textbf{Teorema~#1.}
        \rmfamily
}{
    \end{tcolorbox}
    \medskip
}
