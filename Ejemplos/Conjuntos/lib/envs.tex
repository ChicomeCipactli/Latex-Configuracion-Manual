% \renewenvironment{proof}[1][\textbf{\textit{Demostración}}]{
\newenvironment{proof}[1][\textbf{\textit{Demostración}}]{
    \par\medskip\noindent \textcolor{azul1}{#1.}
    \rmfamily
}{
    $\hfill\textcolor{azul1}{\blacksquare}$\\
    \medskip
}

\newenvironment{solution}[1][\textbf{\textit{Solución}}]{
    \par\medskip\noindent \textcolor{azul1}{#1.}
    \rmfamily
}{
    $\hfill\textcolor{azul1}{\square}$\\
    \medskip
}

{
    \begin{center}
    \begin{tabular}{|p{0.7\textwidth}|}
    \hline
}{ 
    \\
    \hline
    \end{tabular} 
    \end{center}
}

\newcounter{problem}
\newenvironment{problem}[1][\theproblem]{
    \begin{tcolorbox}[colback=blue!20,colframe=blue!50,before upper={\parindent15pt},breakable]
        \refstepcounter{problem}
        % \par\medskip
        \noindent 
        \textbf{Problema~#1.}
        \rmfamily
}{
    \end{tcolorbox}
    \medskip
}

\newcounter{lemma}
\newenvironment{lemma}[1][\thelemma]{
    \begin{tcolorbox}[colback=white!20,colframe=blue!50]
        \refstepcounter{lemma}
        \noindent
        \color[rgb]{0.2,0,0.6}
        \textbf{Lema~#1.}
        \rmfamily
}{
    \end{tcolorbox}
    \medskip
}
