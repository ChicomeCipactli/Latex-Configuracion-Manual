% --------- Definiciones de teoremas, definiciones, axiomas, problemas...

\newenvironment{proof}[1][\textbf{\textit{Demostración}}]{
     \par\medskip\noindent
     \textcolor{tscolor}{#1.}
     \rmfamily
}{
    $\hfill\textcolor{tscolor}{\blacksquare}$\\
}

\newenvironment{solution}[1][\textbf{\textit{Solución}}]{
     \par\medskip\noindent
     \textcolor{tscolor}{#1.}
     \rmfamily
}{
    $\hfill\textcolor{tscolor}{\square}$\\
}

\newcounter{definition}
\newenvironment{definition}[1][\thedefinition]{
    \begin{tcolorbox}[
        colback=white!20,
        colframe=white,
        arc=3mm,
        outer arc=1mm,
        sharp corners=downhill
        ]
        \refstepcounter{definition}
        \noindent
        \color{tscolor}
        \textbf{Definición~#1.}
        \rmfamily
}{
    \end{tcolorbox}
    \medskip
}
